## Loopy
Drawing a bunch of circles is terrible. Let's not do that.

<script type="text/p5" data-autoplay data-preview-width="" data-preview-height="">
function setup() {
  createCanvas(windowWidth,windowHeight);
}

function draw() {
  x = 50;
  while(x<windowWidth){
    ellipse(x,50,50,50);
    x = x + 50;
  }
}
</script>
