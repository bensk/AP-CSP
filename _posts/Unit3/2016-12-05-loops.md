function setup() {
  createCanvas(windowWidth, windowHeight);
}

function draw() {
  x = 50;
  while(x<windowWidth){
    ellipse(x,50,50,50);
    x = x + 50;
  }
}

## `while` loops

Work like a **conditional**:

```javascript
if (something True){
    // do this code
} else {
    // do this code
}
```

```js
while(something True){
    // run this code over and over again
}
```
